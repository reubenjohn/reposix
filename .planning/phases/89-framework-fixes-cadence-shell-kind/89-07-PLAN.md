3. **Create `quality/runners/_audit_field.py`** (mirrors `_freshness.py` shape; stdlib-only):
   ```python
   """RBF-FW-11: claim_vs_assertion_audit + kind:shell-subprocess schema validation.

   Per Decision 3 (COMPLETENESS-CHECK S2): every catalog row minted on/after
   2026-05-08T00:00:00Z MUST carry a `claim_vs_assertion_audit` paragraph
   (≥50 chars) explaining how the verifier's expected.asserts would falsify
   the description claim if false. Runner cross-checks at load time per
   PROTOCOL.md Principle B (fail loud, structured, agent-resolvable).

   REVIEW-FIX MEDIUM-6: rows declaring `kind: shell-subprocess` MUST also
   carry an emitting-transcript contract (expected.artifact.transcript_path
   or schema-equivalent). RBF-FW-11 reads as "every claim falsifiable by an
   assertion"; for kind:shell-subprocess rows the transcript IS the assertion
   shape.

   Date-cutoff design grandfathers the 388 existing P78–P88 rows; P95 RBF-D-06
   backfills them. Sibling module per _freshness.py precedent.
   """
   from __future__ import annotations
   import hashlib

   # REVIEW-FIX LOW-8: use Z suffix (not +00:00) for parser portability.
   # Step 1.5's parser-divergence smoke confirms parse_rfc3339 handles both,
   # but Z is the more compatible form across strict ISO-8601 parsers.
   CUTOFF_ISO = "2026-05-08T00:00:00Z"
   MIN_LEN = 50


   def _row_kind(row: dict) -> str:
       return row.get("kind", "")


   def _has_transcript_contract(row: dict) -> bool:
       """REVIEW-FIX MEDIUM-6: True iff row declares an emitting-transcript artifact.

       Looks for transcript_path under expected.artifact (preferred schema location)
       OR at row-level `transcript_path` OR in `expected.asserts` text mentioning
       'transcript' (last-resort fallback for transitional rows). Sub-subagent
       confirms the actual canonical location based on quality/catalogs/README.md
       after 89-04 lands the kind:shell-subprocess paragraph; tighten this
       function once the canonical field is named.
       """
       expected = row.get("expected", {})
       artifact = expected.get("artifact") if isinstance(expected, dict) else None
       if isinstance(artifact, dict) and artifact.get("transcript_path"):
           return True
       if row.get("transcript_path"):
           return True
       asserts = expected.get("asserts", []) if isinstance(expected, dict) else []
       if any("transcript" in str(a).lower() for a in asserts):
           return True
       return False


   def validate_row(row: dict, catalog_path: str, parse_rfc3339) -> None:
       """Raise SystemExit if row minted ≥ cutoff lacks claim_vs_assertion_audit
       OR if row declares kind:shell-subprocess without an emitting-transcript contract."""
       lv = row.get("last_verified")
       cutoff = parse_rfc3339(CUTOFF_ISO)
       # Rows with last_verified=None (newly minted, never verified) are subject to the check.
       # Rows with last_verified >= cutoff are subject to the check.
       # Rows with last_verified < cutoff are grandfathered (P95 backfills).
       is_new = lv is None or parse_rfc3339(lv) >= cutoff
       if is_new:
           audit = row.get("claim_vs_assertion_audit")
           if not isinstance(audit, str) or len(audit.strip()) < MIN_LEN:
               raise SystemExit(
                   f"FAIL: {catalog_path}: row {row.get('id', '?')} missing "
                   f"claim_vs_assertion_audit (≥{MIN_LEN} chars required for rows "
                   f"minted on/after {CUTOFF_ISO}); see "
                   f"quality/catalogs/README.md schema table for the field's contract"
               )

       # REVIEW-FIX MEDIUM-6: kind:shell-subprocess rows MUST have a transcript contract,
       # regardless of mint date — the kind itself implies a transcript artifact.
       if _row_kind(row) == "shell-subprocess" and not _has_transcript_contract(row):
           raise SystemExit(
               f"FAIL: {catalog_path}: row {row.get('id', '?')} declares "
               f"kind: shell-subprocess but has no transcript-emitting contract "
               f"(expected.artifact.transcript_path or schema-equivalent); see "
               f"quality/catalogs/README.md kind:shell-subprocess paragraph"
           )


   def compute_hash(row: dict) -> str | None:
       """sha256 hex of claim_vs_assertion_audit text (stripped); None if absent."""
       audit = row.get("claim_vs_assertion_audit")
       if not isinstance(audit, str):
           return None
       return hashlib.sha256(audit.strip().encode("utf-8")).hexdigest()
   ```

   class TestKindShellSubprocessTranscriptRule(unittest.TestCase):
       """REVIEW-FIX MEDIUM-6: kind:shell-subprocess rows MUST have transcript contract."""

       def test_shell_subprocess_without_transcript_fails(self):
           row = {
               "id": "x/y",
               "last_verified": "2026-05-08T00:00:00Z",

   class TestComputeHash(unittest.TestCase):
       def test_returns_sha256_hex(self):

           h = _audit_field.compute_hash(row)

           self.assertEqual(len(h), 64)
