2. **Create `quality/runners/_realbackend.py`** (mirrors `_freshness.py` shape; stdlib-only per `quality/runners/run.py:1-30` header):

   ```python
   """Env-gating helper for cadence: pre-release-real-backend.

   Per RBF-FW-01: rows tagged with this cadence MUST have REPOSIX_ALLOWED_ORIGINS
   set to a non-127.0.0.1 origin AND at least one backend credential set complete.
   Default-skips when env unset (NOT-VERIFIED status; never executes against
   missing credentials).

   Sibling module per _freshness.py precedent — keeps run.py under 350-line cap.
   """
   from __future__ import annotations
   import os
   import re
   from typing import Mapping

   _CRED_SETS = {
       "confluence": ("ATLASSIAN_API_KEY", "ATLASSIAN_EMAIL", "REPOSIX_CONFLUENCE_TENANT"),
       "github": ("GITHUB_TOKEN",),
       "jira": ("JIRA_EMAIL", "JIRA_API_TOKEN", "REPOSIX_JIRA_INSTANCE"),
   }
   _NON_LOCAL_RE = re.compile(r"^https?://(?!127\.0\.0\.1)")

   CADENCE_NAME = "pre-release-real-backend"

   # exit-75 → NOT-VERIFIED mapping for verifiers that
   # need to honestly signal "not-verified" via exit code without inventing a
   # bespoke env-gate path. 89-06's SLOT verifier uses this. Follows the BSD
   # sysexits.h convention (EX_TEMPFAIL=75, "temporary failure; user is invited
   # to retry"); we repurpose it as "verifier completed but the conditions for
   # a real grade are not met — preserve NOT-VERIFIED, do NOT overwrite to FAIL".
   EXIT_NOT_VERIFIED = 75


   def is_skipped(row: dict, env: Mapping[str, str]) -> bool:
       """Return True iff row is tagged pre-release-real-backend AND env not configured."""
       if CADENCE_NAME not in row.get("cadences", []):
           return False
       origins = env.get("REPOSIX_ALLOWED_ORIGINS", "")
       if not origins or not _NON_LOCAL_RE.search(origins):
           return True
       return not any(all(env.get(k) for k in keys) for keys in _CRED_SETS.values())


   def skip_reason(env: Mapping[str, str]) -> str:
       """Human-readable reason string for the skip artifact."""
       origins = env.get("REPOSIX_ALLOWED_ORIGINS", "")
       if not origins:
           return "REPOSIX_ALLOWED_ORIGINS unset"
       if not _NON_LOCAL_RE.search(origins):
           return f"REPOSIX_ALLOWED_ORIGINS={origins} matches 127.0.0.1 (local-only)"
       return "no credential set complete (need Confluence: ATLASSIAN_API_KEY+ATLASSIAN_EMAIL+REPOSIX_CONFLUENCE_TENANT, OR GitHub: GITHUB_TOKEN, OR JIRA: JIRA_EMAIL+JIRA_API_TOKEN+REPOSIX_JIRA_INSTANCE)"


   def map_exit_code_to_status(exit_code: int, default_pass: str = "PASS",
                               default_fail: str = "FAIL") -> str:
       """Map verifier exit code to runner status.

       Exit 0  → PASS
       Exit 75 → NOT-VERIFIED (sysexits.h EX_TEMPFAIL repurposed)
       Other   → FAIL

       Use this from run.py's exit-code → status logic so verifiers like 89-06's
       SLOT can exit 75 when the substrate is not landed and have the runner
       preserve NOT-VERIFIED rather than overwriting with FAIL.
       """
       if exit_code == 0:
           return default_pass
       if exit_code == EXIT_NOT_VERIFIED:
           return "NOT-VERIFIED"
       return default_fail
   ```
