```python
#!/usr/bin/env python3
"""quality/gates/docs-build/badges-resolve.py -- BADGE-01 verifier.

HEADs every badge URL extracted from README.md + docs/index.md and asserts
HTTP 200 + Content-Type contains 'image' (or 'json' for the QG-09
endpoint URL).

Exit codes: 0 = PASS (all URLs 200 + correct content-type), 1 = FAIL
(any 4xx/5xx OR wrong content-type), 2 = PARTIAL (transient or
documented Wave-F-pending URLs).

Pre-Wave-F: the QG-09 endpoint badge URL set (WAVE_F_PENDING_URLS) is
skipped with a logged note; the row grades PARTIAL until Wave F lands
the publish + the README badge edit.
"""

from __future__ import annotations

import argparse
import json
import re
import sys
import urllib.error
import urllib.request
from datetime import datetime, timezone
from pathlib import Path

REPO_ROOT = Path(__file__).resolve().parent.parent.parent.parent
ARTIFACT_DIR = REPO_ROOT / "quality" / "reports" / "verifications" / "docs-build"

# Markdown badge image extraction: matches ![<alt>](https://...image-url...)
# Excludes link-target URLs (the OUTER url in [![alt](img)](outer)).
BADGE_IMG_RE = re.compile(r"!\[[^\]]*\]\((https?://[^)]+)\)")

# QG-09 endpoint URLs land in P60 Wave F; until then, skip + grade PARTIAL.
WAVE_F_PENDING_URLS = {
    "https://reubenjohn.github.io/reposix/badge.json",
    # The shields.io endpoint URL that consumes the above also pends Wave F:
    "https://img.shields.io/endpoint?url=https%3A%2F%2Freubenjohn.github.io%2Freposix%2Fbadge.json",
}

TIMEOUT_S = 10


def now_iso() -> str:
    return datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")


def extract_badge_urls(path: Path) -> list[str]:
    """Return deduplicated list of badge image URLs from one markdown file."""
    text = path.read_text(encoding="utf-8")
    seen: list[str] = []
    for match in BADGE_IMG_RE.finditer(text):
        url = match.group(1)
        if url not in seen:
            seen.append(url)
    return seen


def head_url(url: str) -> tuple[int | None, str | None, str | None]:
    """Return (status_code, content_type, error_message)."""
    req = urllib.request.Request(url, method="HEAD",
                                  headers={"User-Agent": "reposix-quality-gates/1.0"})
    try:
        with urllib.request.urlopen(req, timeout=TIMEOUT_S) as resp:
            return resp.status, resp.headers.get("Content-Type", ""), None
    except urllib.error.HTTPError as e:
        return e.code, e.headers.get("Content-Type", "") if e.headers else "", str(e)
    except (urllib.error.URLError, TimeoutError, OSError) as e:
        return None, None, str(e)


def main() -> int:
    parser = argparse.ArgumentParser(description="BADGE-01 verifier")
    parser.add_argument("--row-id", default="docs-build/badges-resolve")
    args = parser.parse_args()

    # Collect URLs from README.md + docs/index.md (deduped).
    urls: list[str] = []
    for src in [REPO_ROOT / "README.md", REPO_ROOT / "docs" / "index.md"]:
        for u in extract_badge_urls(src):
            if u not in urls:
                urls.append(u)

    asserts_passed: list[str] = []
    asserts_failed: list[str] = []
    url_results: dict[str, dict] = {}
    has_partial = False

    for url in urls:
        if url in WAVE_F_PENDING_URLS:
            url_results[url] = {"status": "WAVE-F-PENDING", "note": "URL lands in P60 Wave F (QG-09 publish)"}
            has_partial = True
            continue

        status, ctype, err = head_url(url)
        url_results[url] = {"status": status, "content_type": ctype, "error": err}

        if status != 200:
            asserts_failed.append(f"{url} HEAD => {status} (expected 200) [{err}]")
        elif "image" not in (ctype or "").lower() and "json" not in (ctype or "").lower():
            asserts_failed.append(f"{url} HEAD => Content-Type {ctype!r} (expected image/* or */json)")
        else:
            asserts_passed.append(f"{url} HEAD => 200 + {ctype}")

    # Determine exit code.
    if asserts_failed:
        exit_code = 1  # FAIL
    elif has_partial:
        exit_code = 2  # PARTIAL (Wave F pending)
    else:
        exit_code = 0  # PASS

    artifact = {
        "ts": now_iso(),
        "row_id": args.row_id,
        "exit_code": exit_code,
        "asserts_passed": asserts_passed,
        "asserts_failed": asserts_failed,
        "badge_urls_checked": list(urls),
        "url_results": url_results,
    }

    # Artifact path discriminated by --row-id (so the dimension-native row
    # and the back-compat row write to distinct artifact files per their
    # catalog row.artifact field).
    if args.row_id == "structure/badges-resolve":
        artifact_path = REPO_ROOT / "quality" / "reports" / "verifications" / "structure" / "badges-resolve.json"
    else:
        artifact_path = ARTIFACT_DIR / "badges-resolve.json"
    artifact_path.parent.mkdir(parents=True, exist_ok=True)
    artifact_path.write_text(json.dumps(artifact, indent=2) + "\n", encoding="utf-8")

    print(f"badges-resolve: {len(asserts_passed)} PASS, {len(asserts_failed)} FAIL, "
          f"{len([v for v in url_results.values() if v.get('status')=='WAVE-F-PENDING'])} pending; "
          f"exit={exit_code}")
    for url, res in url_results.items():
        print(f"  {url} => {res}")
    return exit_code


if __name__ == "__main__":
    sys.exit(main())
```

**Network considerations:**

The pre-push runner invokes the verifier on every push. With ~7 HTTP HEADs at ~200-500ms each (cold) → ~3-4s per pre-push run. Per the prompt: "cheap HEAD; ~1s for all 6 badges" — that's optimistic. Use a timeout of 10s per request as a safety margin (TIMEOUT_S in the script). Total worst-case: 7 URLs × 10s = 70s, but typical: <5s. The catalog row's `verifier.timeout_s` is 30s.

If a network outage takes the runner offline, the verifier returns FAIL for affected URLs. That's correct behavior — but the row is P2 so it doesn't block pre-push exit. If badges chronically fail in CI but pass locally, file a SURPRISES.md entry + investigate (DNS? user-agent block? rate limit?).
