**parse_duration signature + impl sketch (target <=15 LOC):**

```python
import re
from datetime import timedelta

_DURATION_RE = re.compile(r"^(\d+)([dhm])$")

def parse_duration(s: str) -> timedelta:
    """Parse a short duration string ('30d', '14d', '90d', optionally 'Xh', 'Xm').

    Raises ValueError on malformed input.
    """
    m = _DURATION_RE.match(s.strip())
    if not m:
        raise ValueError(f"invalid duration: {s!r}")
    n = int(m.group(1))
    unit = m.group(2)
    if unit == "d":
        return timedelta(days=n)
    if unit == "h":
        return timedelta(hours=n)
    return timedelta(minutes=n)  # 'm'
```

**is_stale signature + impl sketch (target <=10 LOC):**

```python
def is_stale(row: dict, now: datetime) -> bool:
    """True iff row.freshness_ttl + row.last_verified < now (and both are present)."""
    ttl = row.get("freshness_ttl")
    lv = row.get("last_verified")
    if not ttl or not lv:
        return False
    try:
        return parse_rfc3339(lv) + parse_duration(ttl) < now
    except ValueError:
        return False
```

"""
import sys
from datetime import datetime, timedelta, timezone
from pathlib import Path

# Add quality/runners to sys.path so we can import run.py directly
sys.path.insert(0, str(Path(__file__).resolve().parent))

from run import parse_duration, is_stale, parse_rfc3339  # noqa


def test_parse_duration_30d():
    assert parse_duration("30d") == timedelta(days=30)


def test_parse_duration_14d():
    assert parse_duration("14d") == timedelta(days=14)


def test_parse_duration_90d():
    assert parse_duration("90d") == timedelta(days=90)


def test_parse_duration_invalid_raises():
    import pytest
    with pytest.raises(ValueError):
        parse_duration("not-a-duration")


def test_is_stale_expired_30d_at_4mo():
    """A row last verified 2026-01-01 with freshness_ttl=30d, evaluated at 2026-04-27, is STALE."""
    row = {"freshness_ttl": "30d", "last_verified": "2026-01-01T00:00:00Z"}
    now = datetime(2026, 4, 27, tzinfo=timezone.utc)
    assert is_stale(row, now) is True


def test_is_stale_within_180d_window():
    """Same row but freshness_ttl=180d, evaluated at 2026-04-27, is NOT STALE."""
    row = {"freshness_ttl": "180d", "last_verified": "2026-01-01T00:00:00Z"}
    now = datetime(2026, 4, 27, tzinfo=timezone.utc)
    assert is_stale(row, now) is False


def test_is_stale_no_last_verified():
    row = {"freshness_ttl": "30d", "last_verified": None}
    now = datetime.now(timezone.utc)
    assert is_stale(row, now) is False


def test_is_stale_no_freshness_ttl():
    row = {"freshness_ttl": None, "last_verified": "2026-01-01T00:00:00Z"}
    now = datetime.now(timezone.utc)
    assert is_stale(row, now) is False
