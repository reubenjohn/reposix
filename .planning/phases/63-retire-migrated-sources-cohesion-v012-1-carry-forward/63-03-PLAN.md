2. Create `quality/gates/code/cargo-fmt-clean.sh`:
   ```bash
   #!/usr/bin/env bash
   # quality/gates/code/cargo-fmt-clean.sh
   #
   # POLISH-CODE final wiring (P63). Verifies row code/cargo-fmt-clean by
   # running `cargo fmt --all -- --check` and writing evidence to
   # quality/reports/verifications/code/cargo-fmt-clean.json.
   #
   # ONE cargo at a time rule (CLAUDE.md): cargo fmt --check is read-only,
   # ~5s, no compile, no link -- safe to run inline.
   set -euo pipefail
   ARTIFACT="quality/reports/verifications/code/cargo-fmt-clean.json"
   mkdir -p "$(dirname "$ARTIFACT")"
   STDOUT=$(mktemp); STDERR=$(mktemp); trap 'rm -f "$STDOUT" "$STDERR"' EXIT
   set +e
   cargo fmt --all -- --check >"$STDOUT" 2>"$STDERR"
   EXIT=$?
   set -e
   STATUS="PASS"; [ $EXIT -ne 0 ] && STATUS="FAIL"
   NOW=$(date -u +%Y-%m-%dT%H:%M:%SZ)
   STDOUT_EX=$(head -c 4000 "$STDOUT" | python3 -c 'import json,sys; print(json.dumps(sys.stdin.read()))')
   STDERR_EX=$(head -c 4000 "$STDERR" | python3 -c 'import json,sys; print(json.dumps(sys.stdin.read()))')
   cat >"$ARTIFACT" <<EOF
   {
     "claim_id": "code/cargo-fmt-clean",
     "phase": "p63",
     "verifier_kind": "mechanical",
     "verified_at": "$NOW",
     "verifier_script": "quality/gates/code/cargo-fmt-clean.sh",
     "asserts": { "cargo_fmt_check_exit_zero": $([ $EXIT -eq 0 ] && echo true || echo false), "no_workspace_drift": $([ $EXIT -eq 0 ] && echo true || echo false) },
     "evidence": { "exit_code": $EXIT, "stdout_excerpt": $STDOUT_EX, "stderr_excerpt": $STDERR_EX },
     "status": "$STATUS"
   }
   EOF
   echo "[cargo-fmt-clean] status=$STATUS exit=$EXIT artifact=$ARTIFACT" >&2
   exit $EXIT
   ```
