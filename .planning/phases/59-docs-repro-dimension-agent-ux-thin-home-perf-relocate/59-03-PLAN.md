**container-rehearse.sh signature + algorithm:**

```bash
#!/usr/bin/env bash
# quality/gates/docs-repro/container-rehearse.sh -- generic container-rehearsal driver.
# Usage: container-rehearse.sh <catalog-row-id>
# Reads quality/catalogs/docs-reproducible.json by row id; runs row.command in row.verifier.container;
# writes artifact to row.artifact with asserts_passed/failed populated.
# <=150 lines. Stdlib bash + jq (or python3 fallback for JSON parsing).
# Graceful skip: if docker not installed, write artifact with status=NOT-VERIFIED + exit 0.

set -uo pipefail  # NOT -e; need to capture docker exit code without bash exiting

ROW_ID="${1:?usage: $0 <catalog-row-id>}"
CATALOG="quality/catalogs/docs-reproducible.json"
ARTIFACT_DIR="quality/reports/verifications/docs-repro"

# 1. Check docker presence
if ! command -v docker >/dev/null 2>&1; then
    # write NOT-VERIFIED artifact + exit 0 (non-fatal per prompt)
    artifact_path=$(python3 -c "import json; rows=json.loads(open('$CATALOG').read())['rows']; r=[x for x in rows if x['id']=='$ROW_ID']; print(r[0]['artifact'] if r else '')")
    mkdir -p "$(dirname "$artifact_path")"
    cat > "$artifact_path" <<EOF
{"ts": "$(date -u +%Y-%m-%dT%H:%M:%SZ)", "row_id": "$ROW_ID", "exit_code": 0, "asserts_passed": [], "asserts_failed": [], "error": "docker not installed; rehearsal skipped (NOT-VERIFIED)"}
EOF
    echo "container-rehearse: docker not found; emitted NOT-VERIFIED artifact" >&2
    exit 0
fi

# 2. Extract command + container from catalog
CATALOG_ROW=$(python3 -c "
import json, sys
rows = json.loads(open('$CATALOG').read())['rows']
r = [x for x in rows if x['id'] == '$ROW_ID']
if not r:
    print('ROW_NOT_FOUND', file=sys.stderr); sys.exit(2)
import json as _j
print(_j.dumps(r[0]))
")
[[ -z "$CATALOG_ROW" ]] && { echo "row not found: $ROW_ID" >&2; exit 1; }

COMMAND=$(echo "$CATALOG_ROW" | python3 -c "import json, sys; r=json.load(sys.stdin); print(r.get('command', ''))")
CONTAINER=$(echo "$CATALOG_ROW" | python3 -c "import json, sys; r=json.load(sys.stdin); print(r.get('verifier', {}).get('container', 'ubuntu:24.04'))")
ARTIFACT_PATH=$(echo "$CATALOG_ROW" | python3 -c "import json, sys; r=json.load(sys.stdin); print(r['artifact'])")

# 3. Run in container with cargo + python3 + curl pre-installed (the example rows assume the workspace built)
SETUP="apt-get update -qq && apt-get install -y -qq curl ca-certificates python3 git build-essential pkg-config libssl-dev > /dev/null"
# For example rows: also need to mount the workspace + build the binaries inside.
# Keep this simple: the host is responsible for `cargo build` first; we mount target/ and PATH.
WORKSPACE="$(pwd)"
docker run --rm \
    -v "$WORKSPACE:/workspace:ro" \
    -v "$WORKSPACE/target:/workspace/target:rw" \
    -w /workspace \
    "$CONTAINER" \
    sh -c "$SETUP && export PATH=/workspace/target/debug:\$PATH && $COMMAND" \
    > /tmp/rehearse-stdout.$$.txt 2> /tmp/rehearse-stderr.$$.txt
EXIT_CODE=$?

# 4. Grade against asserts (per-row asserts.passed/failed populated; hand-graded for now -- v0.12.1 may automate)
# For Wave C v0.12.0: status comes from EXIT_CODE; asserts_passed = the row.expected.asserts entries (assumed met if exit 0); asserts_failed empty if exit 0
# Future v0.12.1 enhancement: parse stdout to verify each individual assert
mkdir -p "$(dirname "$ARTIFACT_PATH")"
cat > "$ARTIFACT_PATH" <<EOF
{
  "ts": "$(date -u +%Y-%m-%dT%H:%M:%SZ)",
  "row_id": "$ROW_ID",
  "exit_code": $EXIT_CODE,
  "container": "$CONTAINER",
  "command": $(printf '%s' "$COMMAND" | python3 -c "import sys, json; print(json.dumps(sys.stdin.read()))"),
  "stdout": $(python3 -c "import json; print(json.dumps(open('/tmp/rehearse-stdout.$$.txt').read()[-2048:]))"),
  "stderr": $(python3 -c "import json; print(json.dumps(open('/tmp/rehearse-stderr.$$.txt').read()[-2048:]))"),
  "asserts_passed": [],
  "asserts_failed": []
}
EOF

rm -f /tmp/rehearse-stdout.$$.txt /tmp/rehearse-stderr.$$.txt
exit $EXIT_CODE
```

**manual-spec-check.sh signature:**

```bash
#!/usr/bin/env bash
# quality/gates/docs-repro/manual-spec-check.sh -- markdown-spec checker for kind: manual rows.
# Usage: manual-spec-check.sh <catalog-row-id>
# For docs-repro/example-03-claude-code-skill: asserts the 2 markdown files exist + reference reposix-agent-flow.
# <=50 lines. Pure stdlib + grep; no docker required.

set -uo pipefail
ROW_ID="${1:?usage: $0 <catalog-row-id>}"
ARTIFACT="quality/reports/verifications/docs-repro/${ROW_ID#docs-repro/}.json"

case "$ROW_ID" in
  docs-repro/example-03-claude-code-skill)
    PASSED=()
    FAILED=()
    for f in examples/03-claude-code-skill/RUN.md examples/03-claude-code-skill/agent-prompt.md; do
      if [[ -s "$f" ]]; then PASSED+=("$f exists with content"); else FAILED+=("$f missing or empty"); fi
    done
    if grep -q '\.claude/skills/reposix-agent-flow' examples/03-claude-code-skill/RUN.md 2>/dev/null; then
      PASSED+=("RUN.md references .claude/skills/reposix-agent-flow")
    else
      FAILED+=("RUN.md does NOT reference .claude/skills/reposix-agent-flow")
    fi
    EXIT_CODE=0
    [[ ${#FAILED[@]} -gt 0 ]] && EXIT_CODE=1
    ;;
  *)
    echo "manual-spec-check: unknown row id $ROW_ID" >&2
    exit 2
    ;;
esac

mkdir -p "$(dirname "$ARTIFACT")"
python3 -c "
import json, sys
artifact = {
  'ts': '$(date -u +%Y-%m-%dT%H:%M:%SZ)',
  'row_id': '$ROW_ID',
  'exit_code': $EXIT_CODE,
  'asserts_passed': $(printf '%s\n' "${PASSED[@]}" | python3 -c 'import json, sys; print(json.dumps([l.strip() for l in sys.stdin if l.strip()]))'),
  'asserts_failed': $(printf '%s\n' "${FAILED[@]}" | python3 -c 'import json, sys; print(json.dumps([l.strip() for l in sys.stdin if l.strip()]))')
}
print(json.dumps(artifact, indent=2))
" > "$ARTIFACT"

exit $EXIT_CODE
```
