```bash
#!/usr/bin/env bash
# P79-02 DVCS-ATTACH-01 — verifier for catalog row
# `agent-ux/reposix-attach-against-vanilla-clone`. TINY shape mirrors
# quality/gates/agent-ux/dark-factory.sh.
#
# Runs: build sim + cli, start sim subprocess, init a vanilla checkout,
# run reposix attach sim::demo, assert extensions.partialClone=reposix and
# remote `reposix` URL contains `reposix::` prefix.
set -euo pipefail
SCRIPT_DIR="$(cd -- "$(dirname -- "${BASH_SOURCE[0]}")" &> /dev/null && pwd)"
REPO_ROOT="$(cd "${SCRIPT_DIR}/../../.." &> /dev/null && pwd)"
cd "${REPO_ROOT}"

PORT=7898
WORK=$(mktemp -d -t reposix-attach-verify.XXXXXX)
trap 'kill ${SIM_PID} 2>/dev/null || true; rm -rf "${WORK}"' EXIT

cargo build -p reposix-sim -p reposix-cli --quiet
target/debug/reposix-sim --bind "127.0.0.1:${PORT}" --ephemeral &
SIM_PID=$!
sleep 1

cd "${WORK}"
git init -q
git remote add origin https://example.invalid/mirror.git  # plain-git mirror placeholder
echo '---' > issues.md
target/debug/reposix attach "sim::demo" --remote-name reposix > /dev/null

PCLONE=$(git config --get extensions.partialClone || true)
RURL=$(git config --get remote.reposix.url || true)
[[ "${PCLONE}" == "reposix" ]] || { echo "FAIL: extensions.partialClone=${PCLONE} expected reposix" >&2; exit 1; }
[[ "${RURL}" == reposix::* ]] || { echo "FAIL: remote.reposix.url=${RURL} expected reposix:: prefix" >&2; exit 1; }
echo "PASS: reposix attach sets extensions.partialClone=reposix and reposix:: remote URL"
exit 0
```

(The `target/debug/reposix` lookup assumes `cargo build -p reposix-cli`
emits the binary at the workspace target dir — confirmed via existing
