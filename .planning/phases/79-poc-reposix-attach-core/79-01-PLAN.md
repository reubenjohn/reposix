`src/main.rs` (~120 lines):

```rust
//! POC reconciler for path (a). Throwaway.
use anyhow::Result;
use clap::Parser;
use reposix_core::frontmatter;
use std::collections::HashMap;
use std::path::PathBuf;
use walkdir::WalkDir;

#[derive(Parser)]
struct Args {
    #[arg(long)]
    working_tree: PathBuf,
    #[arg(long)]
    sot_origin: String,
    #[arg(long)]
    project: String,
}

#[tokio::main]
async fn main() -> Result<()> {
    let args = Args::parse();

    // 1. List backend records via the sim's list_records endpoint.
    let url = format!("{}/projects/{}/records", args.sot_origin, args.project);
    let backend: Vec<serde_json::Value> = reqwest::get(&url).await?.json().await?;
    let backend_ids: HashMap<u64, &serde_json::Value> = backend
        .iter()
        .filter_map(|r| r.get("id").and_then(|v| v.as_u64()).map(|id| (id, r)))
        .collect();

    // 2. Walk the working tree; for each .md file, parse frontmatter; collect
    //    id → [paths] map.
    let mut local_ids: HashMap<u64, Vec<PathBuf>> = HashMap::new();
    let mut no_id_files: Vec<PathBuf> = Vec::new();
    for entry in WalkDir::new(&args.working_tree) {
        let entry = entry?;
        if entry.file_type().is_file() && entry.path().extension().map(|e| e == "md").unwrap_or(false) {
            let bytes = std::fs::read_to_string(entry.path())?;
            match frontmatter::parse(&bytes) {
                Ok(rec) => {
                    let id = rec.id.0 as u64; // RecordId is wrapper
                    local_ids.entry(id).or_default().push(entry.path().to_path_buf());
                }
                Err(_) => {
                    no_id_files.push(entry.path().to_path_buf());
                }
            }
        }
    }

    // 3. Print the reconciliation table.
    println!("=== Reconciliation table (POC) ===");
    let mut duplicate_seen = false;
    let mut backend_deleted_seen = false;
    let mut no_id_seen = false;
    let mut match_seen = false;
    let mut mirror_lag_seen = false;

    for (id, paths) in &local_ids {
        if paths.len() > 1 {
            println!("DUPLICATE_ID id={} files={:?}", id, paths);
            duplicate_seen = true;
        } else if backend_ids.contains_key(id) {
            println!("MATCH id={} file={:?}", id, paths[0]);
            match_seen = true;
        } else {
            println!("BACKEND_DELETED id={} local_file={:?}", id, paths[0]);
            backend_deleted_seen = true;
        }
    }
    for f in &no_id_files {
        println!("NO_ID file={:?}", f);
        no_id_seen = true;
    }
    for backend_id in backend_ids.keys() {
        if !local_ids.contains_key(backend_id) {
            println!("MIRROR_LAG id={} (backend has, local lacks)", backend_id);
            mirror_lag_seen = true;
        }
    }

    let all = match_seen && backend_deleted_seen && no_id_seen && duplicate_seen && mirror_lag_seen;
    println!("\nALL_5_CASES_OBSERVED={}", all);
    std::process::exit(if all { 0 } else { 1 });
}
```

Strategy: spin up a SECOND simulator instance to play the "GH mirror" role

  to express lag without confusing the existing cache state model are
  tagged INFO or REVISE.
- Are there any ordering interleavings where a partial write leaves the
  cache in an unrecoverable state? Specifically: if SoT-write succeeds,
  cache audit row is written, then the helper crashes BEFORE the mirror
  write is attempted, what should the next push do? The POC should EXPLICITLY
  test this scenario (e.g., kill the path-b.sh script between SoT-write and
  mirror-write).
