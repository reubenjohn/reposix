Helper module at the top of the test file:

```rust
struct SimSubprocess {
    child: std::process::Child,
    port: u16,
}

impl SimSubprocess {
    fn start() -> Self {
        // Find a free port (bind to 0, take the assigned port).
        let port = pick_free_port();
        let mut child = Command::new(env!("CARGO_BIN_EXE_reposix-sim"))
            .args(["--bind", &format!("127.0.0.1:{port}"), "--ephemeral"])
            .spawn()
            .unwrap();
        // Wait for sim to bind (busy-poll up to 5s).
        wait_for_bind(port, std::time::Duration::from_secs(5));
        Self { child, port }
    }
}

impl Drop for SimSubprocess {
    fn drop(&mut self) {
        let _ = self.child.kill();
    }
}

fn pick_free_port() -> u16 {
    use std::net::TcpListener;
    TcpListener::bind("127.0.0.1:0").unwrap().local_addr().unwrap().port()
}

fn wait_for_bind(port: u16, timeout: std::time::Duration) {
    let start = std::time::Instant::now();
    loop {
        if std::net::TcpStream::connect(format!("127.0.0.1:{port}")).is_ok() {
            return;
        }
        if start.elapsed() > timeout {
            panic!("sim did not bind on port {port} within {timeout:?}");
        }
        std::thread::sleep(std::time::Duration::from_millis(50));
    }
}

fn git_init(path: &std::path::Path) {
    use std::process::Command;
    Command::new("git").arg("-C").arg(path).args(["init", "-q"]).status().unwrap();
}
```

6. `attach_marks_mirror_lag_for_next_fetch` — backend has id=99 but no
