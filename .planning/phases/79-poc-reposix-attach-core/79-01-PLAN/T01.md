**run.sh skeleton** — see T02-T04 for the actual three paths. T01 just
authors the framing:

```bash
#!/usr/bin/env bash
# POC-01 runner (P79 / v0.13.0). Throwaway end-to-end against the simulator.
# See research/v0.13.0-dvcs/poc/README.md.
set -euo pipefail
SCRIPT_DIR="$(cd -- "$(dirname -- "${BASH_SOURCE[0]}")" &> /dev/null && pwd)"
REPO_ROOT="$(cd "${SCRIPT_DIR}/../../.." &> /dev/null && pwd)"
cd "${REPO_ROOT}"

# Isolation: dedicated cache + sim port; aggressive cleanup on EXIT.
export REPOSIX_CACHE_DIR=/tmp/reposix-poc-79-cache
SIM_PORT="${POC_SIM_PORT:-7888}"  # non-default to avoid clobbering dev sim
SIM_PID=""
cleanup() {
  [[ -n "${SIM_PID}" ]] && kill "${SIM_PID}" 2>/dev/null || true
  rm -rf /tmp/reposix-poc-79-cache /tmp/reposix-poc-79-* 2>/dev/null || true
}
trap cleanup EXIT

mkdir -p "${SCRIPT_DIR}/logs"
echo "POC-01 starting at $(date -u +%Y-%m-%dT%H:%M:%SZ)"

# T02: path (a) — reconciliation against mangled checkout
bash "${SCRIPT_DIR}/path-a.sh" 2>&1 | tee "${SCRIPT_DIR}/logs/path-a-reconciliation.log"

# T03: path (b) — bus SoT-first observing mirror lag
bash "${SCRIPT_DIR}/path-b.sh" 2>&1 | tee "${SCRIPT_DIR}/logs/path-b-bus-mirror-lag.log"

# T04: path (c) — cheap-precheck on SoT mismatch
bash "${SCRIPT_DIR}/path-c.sh" 2>&1 | tee "${SCRIPT_DIR}/logs/path-c-cheap-precheck.log"

echo "POC-01 complete at $(date -u +%Y-%m-%dT%H:%M:%SZ)"
```
