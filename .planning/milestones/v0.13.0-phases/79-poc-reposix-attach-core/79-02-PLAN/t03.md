### 3c. Walker — `crates/reposix-cache/src/reconciliation.rs`

Author the new module (≤ 250 lines):

```rust
//! Reconciliation walk: match working-tree files to backend records by
//! `id` in frontmatter (DVCS-ATTACH-02).
//!
//! Five cases per architecture-sketch.md § "Reconciliation cases":
//!   1. Match: local file's id matches a backend record → row in cache_reconciliation.
//!   2. Backend-deleted: local file's id has no matching backend record → warn + skip.
//!   3. No-id: local file lacks `id` frontmatter → warn + skip.
//!   4. Duplicate-id: two local files claim the same id → hard error.
//!   5. Mirror-lag: backend has id; no local file → cache notes for next fetch.

use std::collections::HashMap;
use std::path::{Path, PathBuf};

use crate::Cache;
use reposix_core::frontmatter;
use reposix_core::RecordId;

/// What `walk_and_reconcile` returns. Cleanly separable cases for testing
/// and for the CLI's stderr summary.
#[derive(Debug, Default)]
pub struct ReconciliationReport {
    pub matched_count: usize,
    pub no_id_count: usize,
    pub backend_deleted_count: usize,
    pub mirror_lag_count: usize,
    pub duplicate_id_files: Vec<(RecordId, Vec<PathBuf>)>,
}

/// Orphan policy controls case-2 (backend-deleted) handling.
#[derive(Debug, Clone, Copy)]
pub enum OrphanPolicy {
    Abort,
    DeleteLocal,
    ForkAsNew,
}

/// Walk the working tree, parse frontmatter, write `cache_reconciliation`
/// rows for matched records. Hard-errors on duplicate-id (case 4). Skips
/// no-id (case 3) and backend-deleted (case 2; subject to OrphanPolicy).
///
/// # Errors
/// - Returns `Err` if the SQLite write fails or duplicate-ids are
///   encountered (no rows committed in the duplicate case — atomic via
///   transaction).
/// - I/O errors during file walk are surfaced.
pub fn walk_and_reconcile(
    work: &Path,
    cache: &mut Cache,
    orphan_policy: OrphanPolicy,
) -> crate::Result<ReconciliationReport> {
    let mut local_ids: HashMap<RecordId, Vec<PathBuf>> = HashMap::new();
    let mut no_id_files: Vec<PathBuf> = Vec::new();

    for entry in walkdir::WalkDir::new(work)
        .into_iter()
        .filter_map(std::result::Result::ok)
    {
        if !entry.file_type().is_file() {
            continue;
        }
        if entry.path().extension().map_or(true, |e| e != "md") {
            continue;
        }
        // Skip files inside `.git/`.
        if entry.path().components().any(|c| c.as_os_str() == ".git") {
            continue;
        }
        let bytes = match std::fs::read_to_string(entry.path()) {
            Ok(b) => b,
            Err(_) => continue,
        };
        match frontmatter::parse(&bytes) {
            Ok(rec) => {
                local_ids.entry(rec.id).or_default().push(entry.path().to_path_buf());
            }
            Err(_) => {
                no_id_files.push(entry.path().to_path_buf());
            }
        }
    }

    let mut report = ReconciliationReport {
        no_id_count: no_id_files.len(),
        ..Default::default()
    };

    // Detect duplicates BEFORE any INSERT (atomicity).
    for (id, paths) in &local_ids {
        if paths.len() > 1 {
            report.duplicate_id_files.push((*id, paths.clone()));
        }
    }
    if !report.duplicate_id_files.is_empty() {
        // Caller handles the bail; we return early without writing.
        return Ok(report);
    }

    // Cross-reference against cache's view of backend records.
    let backend_records = cache.list_record_ids()?;
    let backend_set: std::collections::HashSet<RecordId> =
        backend_records.iter().copied().collect();

    let now = chrono::Utc::now().to_rfc3339();
    let conn = cache.connection_mut()?;
    let tx = conn.transaction()?;
    for (id, paths) in &local_ids {
        let path = &paths[0];
        if backend_set.contains(id) {
            // Case 1: match. Write cache_reconciliation row.
            // (find_oid_for_record requires &Cache, but we hold &mut conn via tx.
            //  Reorder: collect the (id, oid) pairs BEFORE the transaction.)
            // This implementation pattern: do the OID lookups first into a Vec,
            // then start the transaction, then INSERT.
            // Sketch elided; executor implements per the existing crate's
            // borrow-checker-friendly patterns.
            let oid = String::new(); // placeholder — replaced by find_oid_for_record value
            tx.execute(
                "INSERT OR REPLACE INTO cache_reconciliation (record_id, oid, local_path, attached_at) \
                 VALUES (?1, ?2, ?3, ?4)",
                rusqlite::params![id.0, oid, path.to_string_lossy(), now],
            )?;
            report.matched_count += 1;
        } else {
            // Case 2: backend-deleted. Honor orphan_policy.
            match orphan_policy {
                OrphanPolicy::Abort => {
                    eprintln!("BACKEND_DELETED id={} local_file={:?}", id.0, path);
                    report.backend_deleted_count += 1;
                }
                OrphanPolicy::DeleteLocal => {
                    let _ = std::fs::remove_file(path);
                    eprintln!("BACKEND_DELETED id={} local_file={:?} action=DELETED", id.0, path);
                    report.backend_deleted_count += 1;
                }
                OrphanPolicy::ForkAsNew => {
                    eprintln!("BACKEND_DELETED id={} local_file={:?} action=FORK_AS_NEW (TODO P82+)", id.0, path);
                    report.backend_deleted_count += 1;
                }
            }
        }
    }

    // Case 5: backend has, local lacks → mirror-lag.
    for backend_id in backend_set.iter() {
        if !local_ids.contains_key(backend_id) {
            report.mirror_lag_count += 1;
        }
    }

    // Case 3: emit warnings for no-id files.
    for f in &no_id_files {
        eprintln!("NO_ID local_file={:?}", f);
    }

    tx.commit()?;
    Ok(report)
}
```
