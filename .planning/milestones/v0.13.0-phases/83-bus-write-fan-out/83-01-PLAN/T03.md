### 3b. Helper fn — `crates/reposix-cache/src/audit.rs`

```rust
/// Insert `op='helper_push_partial_fail_mirror_lag'` row — one per
/// bus push where SoT writes succeeded but the mirror push
/// subprocess (`git push <mirror_remote_name> main`) failed
/// (DVCS-BUS-WRITE-02 OP-3). The row records the SoT SHA the
/// `refs/mirrors/<sot>-head` ref advanced to, the `git push`
/// subprocess exit code, and a 3-line stderr tail (T-83-02
/// trim — bound the leak surface for operator-readable storage).
///
/// Best-effort: a CHECK violation on a stale cache.db (where the
/// CHECK list does not yet include this op) WARN-logs and the
/// caller's push still succeeds. Fresh caches accept the row
/// immediately. Pitfall 7 / D-03 / D-06.
pub fn log_helper_push_partial_fail_mirror_lag(
    conn: &Connection,
    backend: &str,
    project: &str,
    sot_sha_hex: &str,
    exit_code: i32,
    stderr_tail: &str,
) {
    let reason = format!("exit={exit_code};tail={stderr_tail}");
    let res = conn.execute(
        "INSERT INTO audit_events_cache (ts, op, backend, project, oid, reason) \
         VALUES (?1, 'helper_push_partial_fail_mirror_lag', ?2, ?3, ?4, ?5)",
        params![Utc::now().to_rfc3339(), backend, project, sot_sha_hex, reason],
    );
    if let Err(e) = res {
        warn!(target: "reposix_cache::audit_failure",
              backend, project, exit_code,
              "log_helper_push_partial_fail_mirror_lag failed: {e}");
    }
}
```

```rust
/// Bus-push partial-fail audit row (DVCS-BUS-WRITE-02 OP-3). Called
/// by `bus_handler::handle_bus_export` on the SoT-success-mirror-fail
/// branch. Best-effort: a CHECK violation on stale cache.db
/// WARN-logs without poisoning the helper's `ok refs/heads/main`
/// ack to git.
///
/// `sot_sha_hex` is the `refs/mirrors/<sot>-head` SHA the ref
/// advanced to. `stderr_tail` should be trimmed to <= 3 lines by
/// the caller (T-83-02 — bound operator-readable info leakage).
///
/// # Panics
/// Panics if the cache's `cache.db` mutex is poisoned.
pub fn log_helper_push_partial_fail_mirror_lag(
    &self,
    sot_sha_hex: &str,
    exit_code: i32,
    stderr_tail: &str,
) {
    let conn = self.db.lock().expect("cache.db mutex poisoned");
    crate::audit::log_helper_push_partial_fail_mirror_lag(
        &conn,
        &self.backend_name,
        &self.project,
        sot_sha_hex,
        exit_code,
        stderr_tail,
    );
}
```

```rust
    conn.execute_batch(include_str!("../fixtures/cache_schema.sql"))
        .expect("apply schema");

        &conn,

        "demo",
        "abcdef0123456789",
        128,
        "remote: hook declined / fatal: unable to push",

    let (op, backend, project, oid, reason): (String, String, String, String, String) =
        conn.query_row(
            "SELECT op, backend, project, oid, reason FROM audit_events_cache \
             WHERE op = 'helper_push_partial_fail_mirror_lag'",
            [],
            |r| Ok((r.get(0)?, r.get(1)?, r.get(2)?, r.get(3)?, r.get(4)?)),
        )
        .expect("row should exist");
    assert_eq!(op, "helper_push_partial_fail_mirror_lag");
    assert_eq!(backend, "sim");
    assert_eq!(project, "demo");
    assert_eq!(oid, "abcdef0123456789");
    assert_eq!(
        reason,
        "exit=128;tail=remote: hook declined / fatal: unable to push"
    );

    let upd_err = conn.execute(
        "UPDATE audit_events_cache SET reason = 'tampered' \
         WHERE op = 'helper_push_partial_fail_mirror_lag'",
        [],
    );
    assert!(upd_err.is_err(), "audit_events_cache UPDATE must be blocked by trigger");
}
```
